if not raknet then print("[NotzHub] raknet not found"); return end

-- ── Config persistente ──
local CFG_FILE = "notzhub_desync_cfg.json"
local cfg = { posX = nil, posY = nil, desyncEnabled = true }
pcall(function()
    if isfile and isfile(CFG_FILE) then
        local d = HttpService:JSONDecode(readfile(CFG_FILE))
        if d.posX then cfg.posX = d.posX end
        if d.posY then cfg.posY = d.posY end
        if d.desyncEnabled ~= nil then cfg.desyncEnabled = d.desyncEnabled end
    end
end)
local function saveCfg()
    pcall(function() if writefile then writefile(CFG_FILE, HttpService:JSONEncode(cfg)) end end)
end

local State = { desync=false, stealing=false, timer=0, isDelaying=false, desyncEnabled=cfg.desyncEnabled }
local lastSafePos = nil

-- ── Hooks ──
local function send_hook(packet)
    if not State.stealing or not State.desync or not State.desyncEnabled then return end
    if packet.PacketId == 0x1B then
        local b = packet.AsBuffer
        buffer.writeu32(b,1,0xFFFFFFFF); buffer.writeu32(b,5,0xFFFFFFFF); buffer.writeu32(b,9,0xFFFFFFFF)
        packet:SetData(b)
    end
end
local function recv_hook(packet)
    if not State.stealing or not State.desync or not State.desyncEnabled then return end
    if packet.PacketId == 0x1B or packet.PacketId == 0x86 then packet:Drop() end
end
pcall(function()
    if raknet.add_send_hook then raknet.add_send_hook(send_hook) end
    if raknet.add_recv_hook then raknet.add_recv_hook(recv_hook) end
end)

    local old = (gethui and gethui() or game.CoreGui):FindFirstChild("NotzHub_V12")
    if old then old:Destroy() end
end)

local function updateToggleUI()
    if State.desyncEnabled then
        toggleBtn.Text="DESYNC: ON"; toggleBtn.TextColor3=Color3.fromRGB(0,230,90)
        toggleStroke.Color=Color3.fromRGB(0,200,80)
    else
        toggleBtn.Text="DESYNC: OFF"; toggleBtn.TextColor3=Color3.fromRGB(255,40,40)
        toggleStroke.Color=Color3.fromRGB(200,30,30)
    end
end
updateToggleUI()

-- ── Rollback ──
local function doRollback()
    if lastSafePos and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        lp.Character.HumanoidRootPart.CFrame = CFrame.new(lastSafePos)
    end
end

-- ── Main loop ──
task.spawn(function()
    while true do
        local isStealing = lp:GetAttribute("Stealing") == true

        if isStealing and not State.isDelaying and State.desyncEnabled then
            State.isDelaying = true; State.stealing = true

            doRollback()
            task.wait(0.1)
            State.desync = true

            State.timer = 5
            local blink = true
            while State.timer > 0 and lp:GetAttribute("Stealing") == true and State.desyncEnabled do
                statusText.Text = string.format("%.1fs", State.timer)
                if State.timer > 2 then
                    statusText.TextColor3 = Color3.fromRGB(255,255,255); statusText.Visible = true
                else
                    statusText.TextColor3 = Color3.fromRGB(255,40,40)
                    blink = not blink; statusText.Visible = blink
                end
                task.wait(0.1); State.timer -= 0.1
            end
            statusText.Visible = true
            if lp:GetAttribute("Stealing") == true and State.desyncEnabled then
                State.desync = false
                statusText.Text = "Syncing..."; statusText.TextColor3 = Color3.fromRGB(0,230,90)
                local char = lp.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local cur = hrp.CFrame
                    hrp.CFrame = cur * CFrame.new(0,1000,0)
                    task.wait(0.05); hrp.CFrame = cur
                end
                task.wait(0.15); State.desync = true
            end
            State.isDelaying = false
        elseif not isStealing or not State.desyncEnabled then
            State.stealing = false; State.desync = false; State.isDelaying = false
            statusText.Text = State.desyncEnabled and "Waiting..." or "DISABLED"
            statusText.TextColor3 = State.desyncEnabled and Color3.fromRGB(255,255,255) or Color3.fromRGB(255,40,40)
            statusText.Visible = true
        end
        task.wait(0.1)
    end
end)

-- ── Detecta pegar brainrot na mão → rollback ──
task.spawn(function()
    local lastToolName = nil
    while true do
        local char = lp.Character
        if char then
            local tool = char:FindFirstChildOfClass("Tool")
            local toolName = tool and tool.Name or nil
            if toolName and toolName ~= lastToolName and toolName ~= "Flying Carpet" then
                if lastSafePos then
                    task.wait(0.05)
                    doRollback()
                end
            end
            lastToolName = toolName
        end
        task.wait(0.1)
    end
end)

-- ── Position saver ──
task.spawn(function()
    while true do
        local char = lp.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        if hrp and (not State.stealing or not State.desync) then
            lastSafePos = hrp.Position
        end
        task.wait(0.1)
    end
end)

-- ── Inputs ──
UIS.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if input.KeyCode == Enum.KeyCode.R then doRollback() end
end)
manualBtn.MouseButton1Click:Connect(doRollback)
toggleBtn.MouseButton1Click:Connect(function()
    State.desyncEnabled = not State.desyncEnabled
    cfg.desyncEnabled = State.desyncEnabled
    if not State.desyncEnabled then State.desync=false; State.stealing=false end
    updateToggleUI(); saveCfg()
end)

-- ── Drag + save pos ──
local dragging, dragStart, startPos
main.InputBegan:Connect(function(i)
    if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then
        dragging=true; dragStart=i.Position; startPos=main.Position
    end
end)
UIS.InputChanged:Connect(function(i)
    if dragging and (i.UserInputType==Enum.UserInputType.MouseMovement or i.UserInputType==Enum.UserInputType.Touch) then
        local d=i.Position-dragStart
        main.Position=UDim2.new(startPos.X.Scale,startPos.X.Offset+d.X,startPos.Y.Scale,startPos.Y.Offset+d.Y)
    end
end)
UIS.InputEnded:Connect(function(i)
    if (i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch) and dragging then
        dragging=false
        cfg.posX=main.Position.X.Offset; cfg.posY=main.Position.Y.Offset
        saveCfg()
    end
end)
