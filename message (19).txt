local POS_FILE = "notz_desync_v11_pos.json"

local function saveUIPos(x, y)
    pcall(writefile, POS_FILE, HttpService:JSONEncode({x = x, y = y}))
end

local function loadUIPos()
    local ok, data = pcall(readfile, POS_FILE)
    if ok and data then
        local ok2, parsed = pcall(function() return HttpService:JSONDecode(data) end)
        if ok2 and parsed and parsed.x and parsed.y then
            return parsed.x, parsed.y
        end
    end
    return nil, nil
end

--// DESYNC HOOKS
local function send_hook(packet)
    if not State.stealing or not State.desync or not State.desyncEnabled then
        return
    end
    if packet.PacketId == 0x1B then
        local b = packet.AsBuffer
        buffer.writeu32(b, 1, 0xFFFFFFFF)
        buffer.writeu32(b, 5, 0xFFFFFFFF)
        buffer.writeu32(b, 9, 0xFFFFFFFF)
        packet:SetData(b)
    end
end

local function recv_hook(packet)
    if not State.stealing or not State.desync or not State.desyncEnabled then
        return
    end
    if packet.PacketId == 0x1B or packet.PacketId == 0x86 then
        packet:Drop()
    end
end

pcall(function()
    if raknet.add_send_hook then raknet.add_send_hook(send_hook) end
    if raknet.add_recv_hook then raknet.add_recv_hook(recv_hook) end
end)

--// CORE LOGIC (10 seconds)
task.spawn(function()
    while true do
        local isStealing = lp:GetAttribute("Stealing") == true

        if isStealing and not State.isDelaying and State.desyncEnabled then
            State.isDelaying = true
            State.stealing = true
            State.desync = true

            if lastSafePos and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
                lp.Character.HumanoidRootPart.CFrame = CFrame.new(lastSafePos)
            end

            State.timer = 5
            local blinkState = true

            while State.timer > 0 and lp:GetAttribute("Stealing") == true and State.desyncEnabled do
                statusText.Text = string.format("%.1fs", State.timer)

                if State.timer > 2 then
                    statusText.TextColor3 = Color3.fromRGB(255, 255, 255)
                    statusText.Visible = true
                else
                    statusText.TextColor3 = Color3.fromRGB(255, 40, 40)
                    blinkState = not blinkState
                    statusText.Visible = blinkState
                end

                task.wait(0.1)
                State.timer = State.timer - 0.1
            end

            statusText.Visible = true

            if lp:GetAttribute("Stealing") == true and State.desyncEnabled then
                State.desync = false
                statusText.Text = "Syncing..."
                statusText.TextColor3 = Color3.fromRGB(0, 255, 100)

                local char = lp.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local currentPos = hrp.CFrame
                    hrp.CFrame = currentPos * CFrame.new(0, 1000, 0)
                    task.wait(0.05)
                    hrp.CFrame = currentPos
                end

                task.wait(0.15)
                State.desync = true
            end

            State.isDelaying = false
        elseif not isStealing or not State.desyncEnabled then
            State.stealing = false
            State.desync = false
            State.isDelaying = false
            statusText.Text = State.desyncEnabled and "Waiting..." or "DISABLED"
            statusText.TextColor3 = State.desyncEnabled and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(255, 40, 40)
            statusText.Visible = true
        end

        task.wait(0.1)
    end
end)

--// DRAG + SAVE POSITION
local dragging, dragStart, startPos
main.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = main.Position
    end
end)
UIS.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
UIS.InputEnded:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and dragging then
        dragging = false
        saveUIPos(main.Position.X.Offset, main.Position.Y.Offset)
    end
end)
